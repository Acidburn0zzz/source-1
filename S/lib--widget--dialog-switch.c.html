  80 static void
  81 dialog_switch_goto (GList * dlg)
     /*  [+81 lib/widget/dialog-switch.c] */
  82 {
  83     if (mc_current != dlg)
  84     {
  85         WDialog *old = DIALOG (mc_current->data);
  86 
  87         mc_current = dlg;
  88 
  89         if (old == midnight_dlg)
  90         {
  91             /* switch from panels to another dialog (editor, viewer, etc) */
  92             dialog_switch_pending = TRUE;
  93             dialog_switch_process_pending ();
  94         }
  95         else
  96         {
  97             /* switch from editor, viewer, etc to another dialog */
  98             widget_set_state (WIDGET (old), WST_SUSPENDED, TRUE);
  99 
 100             if (DIALOG (dlg->data) != midnight_dlg)
 101                 /* switch to another editor, viewer, etc */
 102                 /* return to panels before run the required dialog */
 103                 dialog_switch_pending = TRUE;
 104             else
 105             {
 106                 /* switch to panels */
 107                 widget_set_state (WIDGET (midnight_dlg), WST_ACTIVE, TRUE);
 108                 do_refresh ();
 109             }
 110         }
 111     }
 112 }

 133 void
 134 dialog_switch_add (WDialog * h)
     /*  [+134 lib/widget/dialog-switch.c] */
 135 {
 136     GList *dlg;
 137 
 138     dlg = g_list_find (mc_dialogs, h);
 139 
 140     if (dlg != NULL)
 141         mc_current = dlg;
 142     else
 143     {
 144         mc_dialogs = g_list_prepend (mc_dialogs, h);
 145         mc_current = mc_dialogs;

 154 void
 155 dialog_switch_remove (WDialog * h)
     /*  [+155 lib/widget/dialog-switch.c] */
 156 {
 157     GList *this;
 158 
 159     if (DIALOG (mc_current->data) == h)
 160         this = mc_current;
 161     else
 162         this = g_list_find (mc_dialogs, h);
 163 
 164     mc_dialogs = g_list_delete_link (mc_dialogs, this);
 165 
 166     /* adjust current dialog */
 167     if (top_dlg != NULL)
 168         mc_current = g_list_find (mc_dialogs, DIALOG (top_dlg->data));
 169     else
 170         mc_current = mc_dialogs;
 171 
 172     /* resume forced the current screen */
 173     if (mc_current != NULL)
 174         widget_set_state (WIDGET (mc_current->data), WST_ACTIVE, TRUE);
 175 }

 221 void
 222 dialog_switch_list (void)
     /*  [+222 lib/widget/dialog-switch.c] */
 223 {
 224     const size_t dlg_num = g_list_length (mc_dialogs);
 225     int lines, cols;
 226     Listbox *listbox;
 227     GList *h;
 228     int i = 0;
 229     int rv;
 230 
 231     if (mc_global.midnight_shutdown || mc_current == NULL)
 232         return;
 233 
 234     lines = MIN ((size_t) (LINES * 2 / 3), dlg_num);
 235     cols = COLS * 2 / 3;
 236 
 237     listbox = create_listbox_window (lines, cols, _("Screens"), "[Screen selector]");
 238 
 239     for (h = mc_dialogs; h != NULL; h = g_list_next (h))
 240     {
 241         WDialog *dlg;
 242         char *title;
 243 
 244         dlg = DIALOG (h->data);
 245 
 246         if ((dlg != NULL) && (dlg->get_title != NULL))
 247             title = dlg->get_title (dlg, WIDGET (listbox->list)->cols - 2);
 248         else
 249             title = g_strdup ("");
 250 
 251         listbox_add_item (listbox->list, LISTBOX_APPEND_BEFORE, get_hotkey (i++), title, NULL,
 252                           FALSE);
 253 
 254         g_free (title);
 255     }
 256 
 257     listbox_select_entry (listbox->list, dlg_num - 1 - g_list_position (mc_dialogs, mc_current));
 258     rv = run_listbox (listbox);
 259 
 260     if (rv >= 0)
 261     {
 262         h = g_list_nth (mc_dialogs, dlg_num - 1 - rv);
 263         dialog_switch_goto (h);
 264     }
 265 }

 269 int
 270 dialog_switch_process_pending (void)
     /*  [+270 lib/widget/dialog-switch.c] */
 271 {
 272     int ret = 0;
 273 
 274     while (dialog_switch_pending)
 275     {
 276         WDialog *h = DIALOG (mc_current->data);
 277         Widget *wh = WIDGET (h);
 278 
 279         dialog_switch_pending = FALSE;
 280         widget_set_state (wh, WST_SUSPENDED, TRUE);
 281         ret = dlg_run (h);
 282         if (widget_get_state (wh, WST_CLOSED))
 283         {
 284             dlg_destroy (h);
 285 
 286             /* return to panels */
 287             if (mc_global.mc_run_mode == MC_RUN_FULL)
 288             {
 289                 mc_current = g_list_find (mc_dialogs, midnight_dlg);
 290                 mc_event_raise (MCEVENT_GROUP_FILEMANAGER, "update_panels", NULL);
 291             }
 292         }
 293     }
 294 
 295     repaint_screen ();
 296 
 297     return ret;
 298 }

 348 mc_refresh (void)
     /*  [+348 lib/widget/dialog-switch.c] */
 349 {
 350 #ifdef ENABLE_BACKGROUND
 351     if (mc_global.we_are_background)
 352         return;
 353 #endif /* ENABLE_BACKGROUND */

 366 void
 367 dialog_change_screen_size (void)
     /*  [+367 lib/widget/dialog-switch.c] */
 368 {
 369     mc_global.tty.winch_flag = 0;
 370 
 371     tty_change_screen_size ();
 372 
 373 #ifdef HAVE_SLANG
 374     tty_keypad (TRUE);
 375     tty_nodelay (FALSE);
 376 #endif
 377 
 378     /* Inform all suspending dialogs */
 379     dialog_switch_got_winch ();
 380     /* Inform all running dialogs */
 381     g_list_foreach (top_dlg, (GFunc) dlg_resize_cb, NULL);
 382 
 383     /* Now, force the redraw */
 384     repaint_screen ();
