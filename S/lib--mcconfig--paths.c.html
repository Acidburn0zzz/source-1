 137 static void
 138 mc_config_mkdir (const char *directory_name, GError ** mcerror)
     /*  [+138 lib/mcconfig/paths.c] */
 139 {
 140     mc_return_if_error (mcerror);
 141 
 142     if ((!g_file_test (directory_name, G_FILE_TEST_EXISTS | G_FILE_TEST_IS_DIR)) &&
 143         (g_mkdir_with_parents (directory_name, 0700) != 0))
 144         mc_propagate_error (mcerror, 0, _("Cannot create %s directory"), directory_name);

 149 static char *
 150 mc_config_init_one_config_path (const char *path_base, const char *subdir, GError ** mcerror)
     /*  [+150 lib/mcconfig/paths.c] */
 151 {
 152     char *full_path;
 153 
 154     mc_return_val_if_error (mcerror, FALSE);
 155 
 156     full_path = g_build_filename (path_base, subdir, (char *) NULL);
 157     if (g_file_test (full_path, G_FILE_TEST_EXISTS))
 158     {
 159         if (g_file_test (full_path, G_FILE_TEST_IS_DIR))
 160         {
 161             config_dir_present = TRUE;
 162         }
 163         else
 164         {
 165             fprintf (stderr, "%s %s\n", _("FATAL: not a directory:"), full_path);
 166             exit (EXIT_FAILURE);
 167         }
 168     }
 169 
 170     mc_config_mkdir (full_path, mcerror);
 171     if (mcerror != NULL && *mcerror != NULL)
 172         MC_PTR_FREE (full_path);
 173 
 174     return full_path;
 175 }

 187 static void
 188 mc_config_copy (const char *old_name, const char *new_name, GError ** mcerror)
     /*  [+188 lib/mcconfig/paths.c] */
 189 {
 190     mc_return_if_error (mcerror);
 191 
 192     if (g_file_test (old_name, G_FILE_TEST_IS_REGULAR))
 193     {
 194         char *contents = NULL;
 195         size_t length;
 196 
 197         if (g_file_get_contents (old_name, &contents, &length, mcerror))
 198             g_file_set_contents (new_name, contents, length, mcerror);
 199 
 200         g_free (contents);
 201         return;
 202     }
 203 
 204     if (g_file_test (old_name, G_FILE_TEST_IS_DIR))
 205     {
 206 
 207         GDir *dir;
 208         const char *dir_name;
 209 
 210         dir = g_dir_open (old_name, 0, mcerror);
 211         if (dir == NULL)
 212             return;
 213 
 214         if (g_mkdir_with_parents (new_name, 0700) == -1)
 215         {
 216             g_dir_close (dir);
 217             mc_propagate_error (mcerror, 0,
 218                                 _("An error occurred while migrating user settings: %s"),
 219                                 unix_error_string (errno));
 220             return;
 221         }
 222 
 223         while ((dir_name = g_dir_read_name (dir)) != NULL)
 224         {
 225             char *old_name2, *new_name2;
 226 
 227             old_name2 = g_build_filename (old_name, dir_name, (char *) NULL);
 228             new_name2 = g_build_filename (new_name, dir_name, (char *) NULL);
 229             mc_config_copy (old_name2, new_name2, mcerror);
 230             g_free (new_name2);
 231             g_free (old_name2);
 232         }
 233     }
 234 }

 286 void
 287 mc_config_init_config_paths (GError ** mcerror)
     /*  [+287 lib/mcconfig/paths.c] */
 288 {
 289     char *dir;
 290 #if MC_HOMEDIR_XDG == 0
 291     char *defined_userconf_dir;
 292 #else
 293     const char *cdir;
 294 #endif
 295 
 296     mc_return_if_error (mcerror);
 297 
 298     if (xdg_vars_initialized)
 299         return;
 300 
 301     /* init mc_home and homedir if not yet */
 302     (void) mc_config_get_home_dir ();
 303 
 304 #if MC_HOMEDIR_XDG
 305     if (mc_home != NULL)
 306     {
 307         dir = g_build_filename (mc_home, ".config", (char *) NULL);
 308         mc_config_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 309         g_free (dir);
 310 
 311         dir = g_build_filename (mc_home, ".cache", (char *) NULL);
 312         mc_cache_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 313         g_free (dir);
 314 
 315         dir = g_build_filename (mc_home, ".local", "share", (char *) NULL);
 316         mc_data_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 317         g_free (dir);
 318     }
 319     else
 320     {
 321         cdir = g_get_user_config_dir ();
 322         if (cdir != NULL && *cdir != '\0')
 323             mc_config_str = mc_config_init_one_config_path (cdir, MC_USERCONF_DIR, mcerror);
 324         else
 325         {
 326             dir = g_build_filename (homedir, ".config", (char *) NULL);
 327             mc_config_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 328             g_free (dir);
 329         }
 330 
 331         cdir = g_get_user_cache_dir ();
 332         if (cdir != NULL && *cdir != '\0')
 333             mc_cache_str = mc_config_init_one_config_path (cdir, MC_USERCONF_DIR, mcerror);
 334         else
 335         {
 336             dir = g_build_filename (homedir, ".cache", (char *) NULL);
 337             mc_cache_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 338             g_free (dir);
 339         }
 340 
 341         cdir = g_get_user_data_dir ();
 342         if (cdir != NULL && *cdir != '\0')
 343             mc_data_str = mc_config_init_one_config_path (cdir, MC_USERCONF_DIR, mcerror);
 344         else
 345         {
 346             dir = g_build_filename (homedir, ".local", "share", (char *) NULL);
 347             mc_data_str = mc_config_init_one_config_path (dir, MC_USERCONF_DIR, mcerror);
 348             g_free (dir);
 349         }
 350     }
 351 
 352     mc_config_fix_migrated_rules ();
 353 #else /* MC_HOMEDIR_XDG */
 354     defined_userconf_dir = tilde_expand (MC_USERCONF_DIR);
 355     if (g_path_is_absolute (defined_userconf_dir))
 356         dir = defined_userconf_dir;
 357     else
 358     {
 359         g_free (defined_userconf_dir);
 360         dir = g_build_filename (mc_config_get_home_dir (), MC_USERCONF_DIR, (char *) NULL);
 361     }
 362 
 363     mc_data_str = mc_cache_str = mc_config_str = mc_config_init_one_config_path (dir, "", mcerror);
 364 
 365     g_free (dir);
 366 #endif /* MC_HOMEDIR_XDG */
 367 
 368     xdg_vars_initialized = TRUE;
 369 }

 444 gboolean
 445 mc_config_migrate_from_old_place (GError ** mcerror, char **msg)
     /*  [+445 lib/mcconfig/paths.c] */
 446 {
 447     char *old_dir;
 448     size_t rule_index;
 449 
 450     mc_return_val_if_error (mcerror, FALSE);
 451 
 452     if (!mc_config_deprecated_dir_present ())
 453         return FALSE;
 454 
 455     old_dir = mc_config_get_deprecated_path ();
 456 
 457     g_free (mc_config_init_one_config_path (mc_config_str, EDIT_DIR, mcerror));
 458 #if MC_HOMEDIR_XDG
 459     g_free (mc_config_init_one_config_path (mc_cache_str, EDIT_DIR, mcerror));
 460     g_free (mc_config_init_one_config_path (mc_data_str, EDIT_DIR, mcerror));
 461 #endif /* MC_HOMEDIR_XDG */
 462 
 463     mc_return_val_if_error (mcerror, FALSE);
 464 
 465     for (rule_index = 0; mc_config_files_reference[rule_index].old_filename != NULL; rule_index++)
 466     {
 467         char *old_name;
 468         if (*mc_config_files_reference[rule_index].old_filename == '\0')
 469             continue;
 470 
 471         old_name =
 472             g_build_filename (old_dir, mc_config_files_reference[rule_index].old_filename,
 473                               (char *) NULL);
 474 
 475         if (g_file_test (old_name, G_FILE_TEST_EXISTS))
 476         {
 477             char *new_name;
 478             const char *basedir = *mc_config_files_reference[rule_index].new_basedir;
 479             const char *filename = mc_config_files_reference[rule_index].new_filename;
 480 
 481             new_name = g_build_filename (basedir, filename, (char *) NULL);
 482             mc_config_copy (old_name, new_name, mcerror);
 483             g_free (new_name);
 484         }
 485         g_free (old_name);
 486     }
 487 
 488 #if MC_HOMEDIR_XDG
 489     *msg = g_strdup_printf (_("Your old settings were migrated from %s\n"
 490                               "to Freedesktop recommended dirs.\n"
 491                               "To get more info, please visit\n"
 492                               "http://standards.freedesktop.org/basedir-spec/basedir-spec-latest.html"),
 493                             old_dir);
 494 #else /* MC_HOMEDIR_XDG */
 495     *msg = g_strdup_printf (_("Your old settings were migrated from %s\n"
 496                               "to %s\n"), old_dir, mc_config_str);
 497 #endif /* MC_HOMEDIR_XDG */
 498 
 499     g_free (old_dir);
 500 
 501     return TRUE;
 502 }

 511 
 512 char *
 513 mc_config_get_full_path (const char *config_name)
     /*  [+513 lib/mcconfig/paths.c] */
 514 {
 515     size_t rule_index;
 516 
 517     if (config_name == NULL)
 518         return NULL;
 519 
 520     if (!xdg_vars_initialized)
 521         mc_config_init_config_paths (NULL);
 522 
 523     for (rule_index = 0; mc_config_files_reference[rule_index].old_filename != NULL; rule_index++)
 524     {
 525         if (strcmp (config_name, mc_config_files_reference[rule_index].new_filename) == 0)
 526         {
 527             return g_build_filename (*mc_config_files_reference[rule_index].new_basedir,
 528                                      mc_config_files_reference[rule_index].new_filename,
 529                                      (char *) NULL);
 530         }
 531     }
 532     return NULL;
 533 }
