 121 static void
 122 dlg_broadcast_msg_to (WDialog * h, widget_msg_t msg, gboolean reverse, widget_options_t flags)
     /*  [+122 lib/widget/dialog.c] */
 123 {
 124     GList *p, *first;
 125 
 126     if (h->widgets == NULL)
 127         return;
 128 
 129     if (h->current == NULL)
 130         h->current = h->widgets;
 131 
 132     if (reverse)
 133         p = dlg_widget_prev (h, h->current);
 134     else
 135         p = dlg_widget_next (h, h->current);
 136 
 137     first = p;
 138 
 139     do
 140     {
 141         Widget *w = WIDGET (p->data);
 142 
 143         if (reverse)
 144             p = dlg_widget_prev (h, p);
 145         else
 146             p = dlg_widget_next (h, p);
 147 
 148         if ((flags == 0) || ((flags & w->options) != 0))
 149             send_message (w, NULL, msg, 0, NULL);
 150     }
 151     while (first != p);
 152 }

 222 static void
 223 do_select_widget (WDialog * h, GList * w, select_dir_t dir)
     /*  [+223 lib/widget/dialog.c] */
 224 {
 225     Widget *w0 = WIDGET (h->current->data);
 226 
 227     if (!dlg_unfocus (h))
 228         return;
 229 
 230     h->current = w;
 231 
 232     do
 233     {
 234         if (dlg_focus (h))
 235             break;
 236 
 237         switch (dir)
 238         {
 239         case SELECT_EXACT:
 240             h->current = g_list_find (h->widgets, w0);
 241             if (dlg_focus (h))
 242                 return;
 243             /* try find another widget that can take focus */
 244             dir = SELECT_NEXT;
 245             /* fallthrough */
 246         case SELECT_NEXT:
 247             h->current = dlg_widget_next (h, h->current);
 248             break;
 249         case SELECT_PREV:
 250             h->current = dlg_widget_prev (h, h->current);
 251             break;
 252         default:
 253             break;
 254         }
 255     }
 256     while (h->current != w);
 257 
 258     if (widget_overlapped (w0, WIDGET (h->current->data)))
 259     {
 260         send_message (h->current->data, NULL, MSG_DRAW, 0, NULL);
 261         send_message (h->current->data, NULL, MSG_FOCUS, 0, NULL);
 262     }
 263 }

 282 static cb_ret_t
 283 dlg_execute_cmd (WDialog * h, long command)
     /*  [+283 lib/widget/dialog.c] */
 284 {
 285     cb_ret_t ret = MSG_HANDLED;
 286     switch (command)
 287     {
 288     case CK_Ok:
 289         h->ret_value = B_ENTER;
 290         dlg_stop (h);
 291         break;
 292     case CK_Cancel:
 293         h->ret_value = B_CANCEL;
 294         dlg_stop (h);
 295         break;
 296 
 297     case CK_Up:
 298     case CK_Left:
 299         dlg_one_up (h);
 300         break;
 301     case CK_Down:
 302     case CK_Right:
 303         dlg_one_down (h);
 304         break;
 305 
 306     case CK_Help:
 307         {
 308             ev_help_t event_data = { NULL, h->help_ctx };
 309             mc_event_raise (MCEVENT_GROUP_CORE, "help", &event_data);
 310         }
 311         break;
 312 
 313     case CK_Suspend:
 314         mc_event_raise (MCEVENT_GROUP_CORE, "suspend", NULL);
 315         refresh_cmd ();
 316         break;
 317     case CK_Refresh:
 318         refresh_cmd ();
 319         break;
 320 
 321     case CK_ScreenList:
 322         if (!widget_get_state (WIDGET (h), WST_MODAL))
 323             dialog_switch_list ();
 324         else
 325             ret = MSG_NOT_HANDLED;
 326         break;
 327     case CK_ScreenNext:
 328         if (!widget_get_state (WIDGET (h), WST_MODAL))
 329             dialog_switch_next ();
 330         else
 331             ret = MSG_NOT_HANDLED;
 332         break;
 333     case CK_ScreenPrev:
 334         if (!widget_get_state (WIDGET (h), WST_MODAL))
 335             dialog_switch_prev ();
 336         else
 337             ret = MSG_NOT_HANDLED;
 338         break;
 339 
 340     default:
 341         ret = MSG_NOT_HANDLED;
 342     }
 343 
 344     return ret;
 345 }

 383 static int
 384 dlg_mouse_event (WDialog * h, Gpm_Event * event)
     /*  [+384 lib/widget/dialog.c] */
 385 {
 386     Widget *wh = WIDGET (h);
 387 
 388     GList *p;
 389 
 390     /* close the dialog by mouse left click out of dialog area */
 391     if (mouse_close_dialog && (wh->pos_flags & WPOS_FULLSCREEN) == 0
 392         && ((event->buttons & GPM_B_LEFT) != 0) && ((event->type & GPM_DOWN) != 0)
 393         && !mouse_global_in_widget (event, wh))
 394     {
 395         h->ret_value = B_CANCEL;
 396         dlg_stop (h);
 397         return MOU_NORMAL;
 398     }
 399 
 400     if (wh->mouse_callback != NULL)
 401     {
 402         int mou;
 403 
 404         mou = dlg_mouse_translator (event, wh);
 405         if (mou != MOU_UNHANDLED)
 406             return mou;
 407     }
 408 
 409     /* send the event to widgets in reverse Z-order */
 410     p = g_list_last (h->widgets);
 411     do
 412     {
 413         Widget *w = WIDGET (p->data);
 414 
 415         if (!widget_get_state (w, WST_DISABLED) && w->mouse_callback != NULL)
 416         {
 417             /* put global cursor position to the widget */
 418             int ret;
 419 
 420             ret = dlg_mouse_translator (event, w);
 421             if (ret != MOU_UNHANDLED)
 422                 return ret;
 423         }
 424 
 425         p = g_list_previous (p);
 426     }
 427     while (p != NULL);
 428 
 429     return MOU_UNHANDLED;
 430 }

 434 static cb_ret_t
 435 dlg_try_hotkey (WDialog * h, int d_key)
     /*  [+435 lib/widget/dialog.c] */
 436 {
 437     GList *hot_cur;
 438     Widget *current;
 439     cb_ret_t handled;
 440     int c;
 441 
 442     if (h->widgets == NULL)
 443         return MSG_NOT_HANDLED;
 444 
 445     if (h->current == NULL)
 446         h->current = h->widgets;
 447 
 448     /*
 449      * Explanation: we don't send letter hotkeys to other widgets if
 450      * the currently selected widget is an input line
 451      */
 452 
 453     current = WIDGET (h->current->data);
 454 
 455     if (widget_get_state (current, WST_DISABLED))
 456         return MSG_NOT_HANDLED;
 457 
 458     if (widget_get_options (current, WOP_IS_INPUT))
 459     {
 460         /* skip ascii control characters, anything else can valid character in
 461          * some encoding */
 462         if (d_key >= 32 && d_key < 256)
 463             return MSG_NOT_HANDLED;
 464     }
 465 
 466     /* If it's an alt key, send the message */
 467     c = d_key & ~ALT (0);
 468     if (d_key & ALT (0) && g_ascii_isalpha (c))
 469         d_key = g_ascii_tolower (c);
 470 
 471     handled = MSG_NOT_HANDLED;
 472     if (widget_get_options (current, WOP_WANT_HOTKEY))
 473         handled = send_message (current, NULL, MSG_HOTKEY, d_key, NULL);
 474 
 475     /* If not used, send hotkey to other widgets */
 476     if (handled == MSG_HANDLED)
 477         return MSG_HANDLED;
 478 
 479     hot_cur = dlg_widget_next (h, h->current);
 480 
 481     /* send it to all widgets */
 482     while (h->current != hot_cur && handled == MSG_NOT_HANDLED)
 483     {
 484         current = WIDGET (hot_cur->data);
 485 
 486         if (widget_get_options (current, WOP_WANT_HOTKEY)
 487             && !widget_get_state (current, WST_DISABLED))
 488             handled = send_message (current, NULL, MSG_HOTKEY, d_key, NULL);
 489 
 490         if (handled == MSG_NOT_HANDLED)
 491             hot_cur = dlg_widget_next (h, hot_cur);
 492     }
 493 
 494     if (handled == MSG_HANDLED)
 495         do_select_widget (h, hot_cur, SELECT_EXACT);
 496 
 497     return handled;
 498 }

 502 static void
 503 dlg_key_event (WDialog * h, int d_key)
     /*  [+503 lib/widget/dialog.c] */
 504 {
 505     cb_ret_t handled;
 506 
 507     if (h->widgets == NULL)
 508         return;
 509 
 510     if (h->current == NULL)
 511         h->current = h->widgets;
 512 
 513     /* TAB used to cycle */
 514     if (!widget_get_options (WIDGET (h), WOP_WANT_TAB))
 515     {
 516         if (d_key == '\t')
 517         {
 518             dlg_one_down (h);
 519             return;
 520         }
 521         else if ((d_key & ~(KEY_M_SHIFT | KEY_M_CTRL)) == '\t')
 522         {
 523             dlg_one_up (h);
 524             return;
 525         }
 526     }
 527 
 528     /* first can dlg_callback handle the key */
 529     handled = send_message (h, NULL, MSG_KEY, d_key, NULL);
 530 
 531     /* next try the hotkey */
 532     if (handled == MSG_NOT_HANDLED)
 533         handled = dlg_try_hotkey (h, d_key);
 534 
 535     if (handled == MSG_HANDLED)
 536         send_message (h, NULL, MSG_HOTKEY_HANDLED, 0, NULL);
 537     else
 538         /* not used - then try widget_callback */
 539         handled = send_message (h->current->data, NULL, MSG_KEY, d_key, NULL);
 540 
 541     /* not used- try to use the unhandled case */
 542     if (handled == MSG_NOT_HANDLED)
 543         handled = send_message (h, NULL, MSG_UNHANDLED_KEY, d_key, NULL);
 544 
 545     if (handled == MSG_NOT_HANDLED)
 546         handled = dlg_handle_key (h, d_key);
 547 
 548     (void) handled;
 549     send_message (h, NULL, MSG_POST_KEY, d_key, NULL);
 550 }

 554 static void
 555 frontend_dlg_run (WDialog * h)
     /*  [+555 lib/widget/dialog.c] */
 556 {
 557     Widget *wh = WIDGET (h);
 558     Gpm_Event event;
 559 
 560     event.x = -1;
 561 
 562     /* close opened editors, viewers, etc */
 563     if (!widget_get_state (wh, WST_MODAL) && mc_global.midnight_shutdown)
 564     {
 565         send_message (h, NULL, MSG_VALIDATE, 0, NULL);
 566         return;
 567     }
 568 
 569     while (widget_get_state (wh, WST_ACTIVE))
 570     {
 571         int d_key;
 572 
 573         if (mc_global.tty.winch_flag != 0)
 574             dialog_change_screen_size ();
 575 
 576         if (is_idle ())
 577         {
 578             if (idle_hook)
 579                 execute_hooks (idle_hook);
 580 
 581             while (widget_get_state (wh, WST_IDLE) && is_idle ())
 582                 send_message (wh, NULL, MSG_IDLE, 0, NULL);
 583 
 584             /* Allow terminating the dialog from the idle handler */
 585             if (!widget_get_state (wh, WST_ACTIVE))
 586                 break;
 587         }
 588 
 589         update_cursor (h);
 590 
 591         /* Clear interrupt flag */
 592         tty_got_interrupt ();
 593         d_key = tty_get_event (&event, h->mouse_status == MOU_REPEAT, TRUE);
 594 
 595         dlg_process_event (h, d_key, &event);
 596 
 597         if (widget_get_state (wh, WST_CLOSED))
 598             send_message (h, NULL, MSG_VALIDATE, 0, NULL);
 599     }
 600 }

 615 static void
 616 dlg_set_top_or_bottom_widget (void *w, gboolean set_top)
     /*  [+616 lib/widget/dialog.c] */
 617 {
 618     Widget *widget = WIDGET (w);
 619     WDialog *h = widget->owner;
 620     GList *l;
 621 
 622     l = g_list_find (h->widgets, w);
 623     if (l == NULL)
 624         abort ();               /* widget is not in dialog, this should not happen */
 625 
 626     /* unfocus prevoius widget and focus current one before widget reordering */
 627     if (set_top && widget_get_state (WIDGET (h), WST_ACTIVE))
 628         do_select_widget (h, l, SELECT_EXACT);
 629 
 630     /* widget reordering */
 631     h->widgets = g_list_remove_link (h->widgets, l);
 632     if (set_top)
 633         h->widgets = g_list_concat (h->widgets, l);
 634     else
 635         h->widgets = g_list_concat (l, h->widgets);
 636 }

 698     shift_x = wh->x - ox;
 699     shift_y = wh->y - oy;
 700     scale_x = wh->cols - oc;
 701     scale_y = wh->lines - ol;

 709             /* there are, mainly, 2 generally possible
 710                situations:

 712                1. control sticks to one side - it
 713                should be moved
 714 
 715                2. control sticks to two sides of
 716                one direction - it should be sized */
 717 
 718             Widget *c = WIDGET (w->data);
 719             int cx = c->x;
 720             int cy = c->y;
 721             int ccols = c->cols;
 722             int clines = c->lines;

 724             if ((c->pos_flags & WPOS_CENTER_HORZ) != 0)
 725                 cx = wh->x + (wh->cols - c->cols) / 2;
 726             else if ((c->pos_flags & WPOS_KEEP_LEFT) != 0 && (c->pos_flags & WPOS_KEEP_RIGHT) != 0)
 727             {
 728                 cx += shift_x;
 729                 ccols += scale_x;
 730             }
 731             else if ((c->pos_flags & WPOS_KEEP_LEFT) != 0)
 732                 cx += shift_x;
 733             else if ((c->pos_flags & WPOS_KEEP_RIGHT) != 0)
 734                 cx += shift_x + scale_x;
 735 
 736             if ((c->pos_flags & WPOS_CENTER_VERT) != 0)
 737                 cy = wh->y + (wh->lines - c->lines) / 2;
 738             else if ((c->pos_flags & WPOS_KEEP_TOP) != 0 && (c->pos_flags & WPOS_KEEP_BOTTOM) != 0)
 739             {
 740                 cy += shift_y;
 741                 clines += scale_y;
 742             }
 743             else if ((c->pos_flags & WPOS_KEEP_TOP) != 0)
 744                 cy += shift_y;
 745             else if ((c->pos_flags & WPOS_KEEP_BOTTOM) != 0)
 746                 cy += shift_y + scale_y;
 747 
 748             widget_set_size (c, cy, cx, clines, ccols);
 749         }

 927 /**
 928  * Insert widget to dialog before requested widget. Make the widget current. Return widget ID.
 929  */
 930 
 931 unsigned long
 932 add_widget_autopos (WDialog * h, void *w, widget_pos_flags_t pos_flags, const void *before)
     /*  [+932 lib/widget/dialog.c] */
 933 {
 934     Widget *wh = WIDGET (h);
 935     Widget *widget;
 936 
 937     /* Don't accept 0 widgets */
 938     if (w == NULL)
 939         abort ();
 940 
 941     widget = WIDGET (w);
 942 
 943     if ((pos_flags & WPOS_CENTER_HORZ) != 0)
 944         widget->x = (wh->cols - widget->cols) / 2;
 945     widget->x += wh->x;
 946 
 947     if ((pos_flags & WPOS_CENTER_VERT) != 0)
 948         widget->y = (wh->lines - widget->lines) / 2;

 957         h->widgets = g_list_append (h->widgets, widget);
 958         h->current = g_list_last (h->widgets);
 959     }
 960     else
 961     {
 962         GList *b;
 963 
 964         b = g_list_find (h->widgets, before);
 965 
 966         /* don't accept widget not from dialog. This shouldn't happen */
 967         if (b == NULL)
 968             abort ();

 973             h->current = g_list_previous (b);
 974         else
 975             h->current = g_list_last (h->widgets);
 976     }
 977 
 978     /* widget has been added in runtime */
 979     if (widget_get_state (wh, WST_ACTIVE))
 980     {
 981         send_message (widget, NULL, MSG_INIT, 0, NULL);
 982         send_message (widget, NULL, MSG_DRAW, 0, NULL);
 983         send_message (widget, NULL, MSG_FOCUS, 0, NULL);
 984     }
 985
