 231 static void
 232 edit_window_move (WEdit * edit, long command)
     /*  [+232 src/editor/editwidget.c] */
 233 {
 234     Widget *w = WIDGET (edit);
 235     Widget *wh = WIDGET (w->owner);
 236 
 237     switch (command)
 238     {
 239     case CK_Up:
 240         if (w->y > wh->y + 1)   /* menubar */
 241             w->y--;
 242         break;
 243     case CK_Down:
 244         if (w->y < wh->y + wh->lines - 2)       /* buttonbar */
 245             w->y++;
 246         break;
 247     case CK_Left:
 248         if (w->x + w->cols > wh->x)
 249             w->x--;
 250         break;
 251     case CK_Right:
 252         if (w->x < wh->x + wh->cols)
 253             w->x++;
 254         break;
 255     default:
 256         return;
 257     }
 258 
 259     edit->force |= REDRAW_PAGE;
 260     dlg_redraw (w->owner);
 261 }

 285             w->lines++;
 286         break;
 287     case CK_Left:
 288         if (w->cols > WINDOW_MIN_COLS)
 289             w->cols--;
 290         break;
 291     case CK_Right:
 292         if (w->x + w->cols < wh->x + wh->cols)
 293             w->cols++;
 294         break;
 295     default:
 296         return;
 297     }
 298 
 299     edit->force |= REDRAW_COMPLETELY;
 300     dlg_redraw (w->owner);

 319 static void
 320 edit_window_list (const WDialog * h)
     /*  [+320 src/editor/editwidget.c] */
 321 {
 322     const size_t offset = 2;    /* skip menu and buttonbar */
 323     const size_t dlg_num = g_list_length (h->widgets) - offset;
 324     int lines, cols;
 325     Listbox *listbox;
 326     GList *w;
 327     int i = 0;
 328     int rv;
 329 
 330     lines = MIN ((size_t) (LINES * 2 / 3), dlg_num);
 331     cols = COLS * 2 / 3;
 332 
 333     listbox = create_listbox_window (lines, cols, _("Open files"), "[Open files]");
 334 
 335     for (w = h->widgets; w != NULL; w = g_list_next (w))
 336         if (edit_widget_is_editor (CONST_WIDGET (w->data)))
 337         {
 338             WEdit *e = (WEdit *) w->data;
 339             char *fname;
 340 
 341             if (e->filename_vpath == NULL)
 342                 fname = g_strdup_printf ("%c [%s]", e->modified ? '*' : ' ', _("NoName"));
 343             else
 344                 fname =
 345                     g_strdup_printf ("%c%s", e->modified ? '*' : ' ',
 346                                      vfs_path_as_str (e->filename_vpath));
 347 
 348             listbox_add_item (listbox->list, LISTBOX_APPEND_AT_END, get_hotkey (i++),
 349                               str_term_trim (fname, WIDGET (listbox->list)->cols - 2), NULL, FALSE);
 350             g_free (fname);
 351         }
 352 
 353     rv = g_list_position (h->widgets, h->current) - offset;
 354     listbox_select_entry (listbox->list, rv);
 355     rv = run_listbox (listbox);
 356     if (rv >= 0)
 357     {
 358         w = g_list_nth (h->widgets, rv + offset);
 359         dlg_select_widget (w->data);
 360     }
 361 }

 409 static cb_ret_t
 410 edit_dialog_command_execute (WDialog * h, long command)
     /*  [+410 src/editor/editwidget.c] */
 411 {
 412     Widget *wh = WIDGET (h);
 413     cb_ret_t ret = MSG_HANDLED;
 414 
 415     switch (command)
 416     {
 417     case CK_EditNew:
 418         edit_add_window (h, wh->y + 1, wh->x, wh->lines - 2, wh->cols, NULL, 0);
 419         break;
 420     case CK_EditFile:
 421         edit_load_cmd (h);
 422         break;
 423     case CK_EditSyntaxFile:
 424         edit_load_syntax_file (h);
 425         break;
 426     case CK_EditUserMenu:
 427         edit_load_menu_file (h);
 428         break;
 429     case CK_Close:
 430         /* if there are no opened files anymore, close MC editor */
 431         if (edit_widget_is_editor (CONST_WIDGET (h->current->data)) &&
 432             edit_close_cmd ((WEdit *) h->current->data) && find_editor (h) == NULL)
 433             dlg_stop (h);
 434         break;
 435     case CK_Help:
 436         edit_help ();
 437         /* edit->force |= REDRAW_COMPLETELY; */
 438         break;
 439     case CK_Menu:
 440         edit_menu_cmd (h);
 441         break;
 442     case CK_Quit:
 443     case CK_Cancel:
 444         /* don't close editor due to SIGINT, but stop move/resize window */
 445         {
 446             Widget *w = WIDGET (h->current->data);
 447 
 448             if (edit_widget_is_editor (w) && ((WEdit *) w)->drag_state != MCEDIT_DRAG_NONE)
 449                 edit_restore_size ((WEdit *) w);
 450             else if (command == CK_Quit)
 451                 dlg_stop (h);
 452         }
 453         break;
 454     case CK_About:
 455         edit_about ();
 456         break;
 457     case CK_SyntaxOnOff:
 458         edit_syntax_onoff_cmd (h);
 459         break;
 460     case CK_ShowTabTws:
 461         edit_show_tabs_tws_cmd (h);
 462         break;
 463     case CK_ShowMargin:
 464         edit_show_margin_cmd (h);
 465         break;
 466     case CK_ShowNumbers:
 467         edit_show_numbers_cmd (h);
 468         break;
 469     case CK_Refresh:
 470         edit_refresh_cmd ();
 471         break;
 472     case CK_Shell:
 473         view_other_cmd ();
 474         break;
 475     case CK_LearnKeys:
 476         learn_keys ();
 477         break;
 478     case CK_WindowMove:
 479     case CK_WindowResize:
 480         if (edit_widget_is_editor (CONST_WIDGET (h->current->data)))
 481             edit_handle_move_resize ((WEdit *) h->current->data, command);
 482         break;
 483     case CK_WindowList:
 484         edit_window_list (h);
 485         break;
 486     case CK_WindowNext:
 487         dlg_one_down (h);
 488         dlg_select_widget (h->current->data);
 489         break;
 490     case CK_WindowPrev:
 491         dlg_one_up (h);
 492         dlg_select_widget (h->current->data);
 493         break;
 494     case CK_Options:
 495         edit_options_dialog (h);
 496         break;
 497     case CK_OptionsSaveMode:
 498         edit_save_mode_cmd ();
 499         break;
 500     case CK_SaveSetup:
 501         save_setup_cmd ();
 502         break;
 503     default:
 504         ret = MSG_NOT_HANDLED;
 505         break;
 506     }
 507 
 508     return ret;
 509 }

 512 /*
 513  * Translate the keycode into either 'command' or 'char_for_insertion'.
 514  * 'command' is one of the editor commands from cmddef.h.
 515  */
 516 
 517 static gboolean
 518 edit_translate_key (WEdit * edit, long x_key, int *cmd, int *ch)
     /*  [+518 src/editor/editwidget.c] */
 519 {
 520     long command = CK_InsertChar;
 521     int char_for_insertion = -1;
 522 
 523     /* an ordinary insertable character */
 524     if (!edit->extmod && x_key < 256)
 525     {
 526 #ifndef HAVE_CHARSET
 527         if (is_printable (x_key))
 528         {
 529             char_for_insertion = x_key;
 530             goto fin;
 531         }
 532 #else
 533         int c;
 534 
 535         if (edit->charpoint >= 4)
 536         {
 537             edit->charpoint = 0;
 538             edit->charbuf[edit->charpoint] = '\0';
 539         }
 540         if (edit->charpoint < 4)
 541         {
 542             edit->charbuf[edit->charpoint++] = x_key;
 543             edit->charbuf[edit->charpoint] = '\0';
 544         }
 545 
 546         /* input from 8-bit locale */
 547         if (!mc_global.utf8_display)
 548         {
 549             /* source in 8-bit codeset */
 550             c = convert_from_input_c (x_key);
 551 
 552             if (is_printable (c))
 553             {
 554                 if (!edit->utf8)
 555                     char_for_insertion = c;
 556                 else
 557                     char_for_insertion = convert_from_8bit_to_utf_c2 ((char) x_key);
 558                 goto fin;
 559             }
 560         }
 561         else
 562         {
 563             /* UTF-8 locale */
 564             int res;
 565 
 566             res = str_is_valid_char (edit->charbuf, edit->charpoint);
 567             if (res < 0 && res != -2)
 568             {
 569                 edit->charpoint = 0;    /* broken multibyte char, skip */
 570                 goto fin;
 571             }
 572 
 573             if (edit->utf8)
 574             {
 575                 /* source in UTF-8 codeset */
 576                 if (res < 0)
 577                 {
 578                     char_for_insertion = x_key;
 579                     goto fin;
 580                 }
 581 
 582                 edit->charbuf[edit->charpoint] = '\0';
 583                 edit->charpoint = 0;
 584                 if (g_unichar_isprint (g_utf8_get_char (edit->charbuf)))
 585                 {
 586                     char_for_insertion = x_key;
 587                     goto fin;
 588                 }
 589             }
 590             else
 591             {
 592                 /* 8-bit source */
 593                 if (res < 0)
 594                 {
 595                     /* not finised multibyte input (in meddle multibyte utf-8 char) */
 596                     goto fin;
 597                 }
 598 
 599                 if (g_unichar_isprint (g_utf8_get_char (edit->charbuf)))
 600                 {
 601                     c = convert_from_utf_to_current (edit->charbuf);
 602                     edit->charbuf[0] = '\0';
 603                     edit->charpoint = 0;
 604                     char_for_insertion = c;
 605                     goto fin;
 606                 }
 607 
 608                 /* unprinteble utf input, skip it */
 609                 edit->charbuf[0] = '\0';
 610                 edit->charpoint = 0;
 611             }
 612         }
 613 #endif /* HAVE_CHARSET */
 614     }
 615 
 616     /* Commands specific to the key emulation */
 617     if (edit->extmod)
 618     {
 619         edit->extmod = FALSE;
 620         command = keybind_lookup_keymap_command (editor_x_map, x_key);
 621     }
 622     else
 623         command = keybind_lookup_keymap_command (editor_map, x_key);
 624 
 625     if (command == CK_IgnoreKey)
 626         command = CK_InsertChar;
 627 
 628   fin:
 629     *cmd = (int) command;       /* FIXME */
 630     *ch = char_for_insertion;
 631 
 632     return !(command == CK_InsertChar && char_for_insertion == -1);
 633 }

 638 static inline void
 639 edit_quit (WDialog * h)
     /*  [+639 src/editor/editwidget.c] */
 640 {
 641     GList *l;
 642     WEdit *e = NULL;
 643 
 644     /* don't stop the dialog before final decision */
 645     widget_set_state (WIDGET (h), WST_ACTIVE, TRUE);
 646 
 647     for (l = h->widgets; l != NULL; l = g_list_next (l))
 648         if (edit_widget_is_editor (CONST_WIDGET (l->data)))
 649         {
 650             e = (WEdit *) l->data;
 651 
 652             if (e->drag_state != MCEDIT_DRAG_NONE)
 653             {
 654                 edit_restore_size (e);
 655                 return;
 656             }
 657 
 658             if (e->modified)
 659             {
 660                 dlg_select_widget (e);
 661 
 662                 if (!edit_ok_to_exit (e))
 663                     return;
 664             }
 665         }
 666 
 667     /* no editors in dialog at all or no any file required to be saved */
 668     if (e == NULL || l == NULL)
 669         dlg_stop (h);
 670 }

 702 static gboolean
 703 edit_update_cursor (WEdit * edit, const mouse_event_t * event)
     /*  [+703 src/editor/editwidget.c] */
 704 {
 705     int x, y;
 706     gboolean done;
 707 
 708     x = event->x - (edit->fullscreen ? 0 : 1);
 709     y = event->y - (edit->fullscreen ? 0 : 1);
 710 
 711     if (edit->mark2 != -1 && event->msg == MSG_MOUSE_UP)
 712         return TRUE;            /* don't do anything */
 713 
 714     if (event->msg == MSG_MOUSE_DOWN || event->msg == MSG_MOUSE_UP)
 715         edit_push_key_press (edit);
 716 
 717     if (!option_cursor_beyond_eol)
 718         edit->prev_col = x - edit->start_col - option_line_state_width;
 719     else
 720     {
 721         long line_len;
 722 
 723         line_len =
 724             edit_move_forward3 (edit, edit_buffer_get_current_bol (&edit->buffer), 0,
 725                                 edit_buffer_get_current_eol (&edit->buffer));
 726 
 727         if (x > line_len - 1)
 728         {
 729             edit->over_col = x - line_len - edit->start_col - option_line_state_width;
 730             edit->prev_col = line_len;
 731         }
 732         else
 733         {
 734             edit->over_col = 0;
 735             edit->prev_col = x - option_line_state_width - edit->start_col;
 736         }
 737     }
 738 
 739     if (y > edit->curs_row)
 740         edit_move_down (edit, y - edit->curs_row, FALSE);
 741     else if (y < edit->curs_row)
 742         edit_move_up (edit, edit->curs_row - y, FALSE);
 743     else
 744         edit_move_to_prev_col (edit, edit_buffer_get_current_bol (&edit->buffer));
 745 
 746     if (event->msg == MSG_MOUSE_CLICK)
 747     {
 748         edit_mark_cmd (edit, TRUE);     /* reset */
 749         edit->highlight = 0;
 750     }
 751 
 752     done = (event->msg != MSG_MOUSE_DRAG);
 753     if (done)
 754         edit_mark_cmd (edit, FALSE);
 755 
 756     return done;
 757 }

 772         edit_dlg_init ();

 774 
 775     case MSG_DRAW:

 778         dlg_erase (h);
 779         return MSG_HANDLED;
 780 
 781     case MSG_RESIZE:

 858 
 859     default:
 860         return dlg_default_callback (w, sender, msg, parm, data);

 995 
 996     case MSG_DESTROY:

 998         return MSG_HANDLED;

1007 /**
1008  * Handle move/resize mouse events.
1009  */
1010 static void
1011 edit_mouse_handle_move_resize (Widget * w, mouse_msg_t msg, mouse_event_t * event)
     /*  [+1011 src/editor/editwidget.c] */
1012 {
1013     WEdit *edit = (WEdit *) (w);
1014     Widget *h = WIDGET (w->owner);
1015     int global_x, global_y;
1016 
1017     if (msg == MSG_MOUSE_UP)
1018     {
1019         /* Exit move/resize mode. */
1020         edit_execute_cmd (edit, CK_Enter, -1);
1021         return;
1022     }
1023 
1024     if (msg != MSG_MOUSE_DRAG)
1025         /**
1026          * We ignore any other events. Specifically, MSG_MOUSE_DOWN.
1027          *
1028          * When the move/resize is initiated by the menu, we let the user
1029          * stop it by clicking with the mouse. Which is why we don't want
1030          * a mouse down to affect the window.
1031          */
1032         return;
1033 
1034     /* Convert point to global coordinates for easier calculations. */
1035     global_x = event->x + w->x;
1036     global_y = event->y + w->y;
1037 
1038     /* Clamp the point to the dialog's client area. */
1039     global_y = CLAMP (global_y, h->y + 1, h->y + h->lines - 2); /* Status line, buttonbar */
1040     global_x = CLAMP (global_x, h->x, h->x + h->cols - 1);      /* Currently a no-op, as the dialog has no left/right margins. */
1041 
1042     if (edit->drag_state == MCEDIT_DRAG_MOVE)
1043     {
1044         w->y = global_y;
1045         w->x = global_x - edit->drag_state_start;
1046     }
1047     else if (edit->drag_state == MCEDIT_DRAG_RESIZE)
1048     {
1049         w->lines = MAX (WINDOW_MIN_LINES, global_y - w->y + 1);
1050         w->cols = MAX (WINDOW_MIN_COLS, global_x - w->x + 1);
1051     }
1052 
1053     edit->force |= REDRAW_COMPLETELY;   /* Not really needed as WEdit's MSG_DRAW already does this. */
1054 
1055     /* We draw the whole dialog because dragging/resizing exposes area beneath. */
1056     dlg_redraw (w->owner);
1057 }

1061 /**
1062  * Handle mouse events of editor window
1063  *
1064  * @param w Widget object (the editor window)
1065  * @param msg mouse event message
1066  * @param event mouse event data
1067  */
1068 static void
1069 edit_mouse_callback (Widget * w, mouse_msg_t msg, mouse_event_t * event)
     /*  [+1069 src/editor/editwidget.c] */
1070 {
1071     WEdit *edit = (WEdit *) w;
1072     /* offset for top line */
1073     int dx = edit->fullscreen ? 0 : 2;
1074     /* location of 'Close' and 'Toggle fullscreen' pictograms */
1075     int close_x, toggle_fullscreen_x;
1076 
1077     close_x = (w->cols - 1) - dx - 1;
1078     toggle_fullscreen_x = close_x - 3;
1079 
1080     if (edit->drag_state != MCEDIT_DRAG_NONE)
1081     {
1082         /* window is being resized/moved */
1083         edit_mouse_handle_move_resize (w, msg, event);
1084         return;
1085     }
1086 
1087     switch (msg)
1088     {
1089     case MSG_MOUSE_DOWN:
1090         dlg_select_widget (w);
1091         edit_update_curs_row (edit);
1092         edit_update_curs_col (edit);
1093 
1094         if (!edit->fullscreen)
1095         {
1096             if (event->y == 0)
1097             {
1098                 if (event->x == close_x)
1099                     ;           /* do nothing (see MSG_MOUSE_CLICK) */
1100                 else if (event->x == toggle_fullscreen_x)
1101                     ;           /* do nothing (see MSG_MOUSE_CLICK) */
1102                 else
1103                 {
1104                     /* start window move */
1105                     edit_execute_cmd (edit, CK_WindowMove, -1);
1106                     edit->drag_state_start = event->x;
1107                 }
1108                 break;
1109             }
1110 
1111             if (event->y == w->lines - 1 && event->x == w->cols - 1)
1112             {
1113                 /* bottom-right corner -- start window resize */
1114                 edit_execute_cmd (edit, CK_WindowResize, -1);
1115                 break;
1116             }
1117         }
1118 
1119         /* fall through to start/stop text selection */
1120 
1121     case MSG_MOUSE_UP:
1122         edit_update_cursor (edit, event);
1123         edit_total_update (edit);
1124         break;
1125 
1126     case MSG_MOUSE_CLICK:
1127         if (event->y == 0)
1128         {
1129             if (event->x == close_x)
1130                 send_message (w->owner, NULL, MSG_ACTION, CK_Close, NULL);
1131             else if (event->x == toggle_fullscreen_x)
1132                 edit_toggle_fullscreen (edit);
1133             else if (!edit->fullscreen && event->count == GPM_DOUBLE)
1134                 /* double click on top line (toggle fullscreen) */
1135                 edit_toggle_fullscreen (edit);
1136         }
1137         else if (event->count == GPM_DOUBLE)
1138         {
1139             /* double click */
1140             edit_mark_current_word_cmd (edit);
1141             edit_total_update (edit);
1142         }
1143         else if (event->count == GPM_TRIPLE)
1144         {
1145             /* triple click: works in GPM only, not in xterm */
1146             edit_mark_current_line_cmd (edit);
1147             edit_total_update (edit);
1148         }
1149         break;
1150 
1151     case MSG_MOUSE_DRAG:
1152         edit_update_cursor (edit, event);
1153         edit_total_update (edit);
1154         break;
1155 
1156     case MSG_MOUSE_SCROLL_UP:
1157         edit_move_up (edit, 2, TRUE);
1158         edit_total_update (edit);
1159         break;
1160 
1161     case MSG_MOUSE_SCROLL_DOWN:
1162         edit_move_down (edit, 2, TRUE);
1163         edit_total_update (edit);
1164         break;
1165 
1166     default:
1167         break;
1168     }
1169 }
