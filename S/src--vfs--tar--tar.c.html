 233 /**
 234  * Quick and dirty octal conversion.
 235  *
 236  * Result is -1 if the field is invalid (all blank, or nonoctal).
 237  */
 238 static long
 239 tar_from_oct (int digs, char *where)
     /*  [+239 src/vfs/tar/tar.c] */
 240 {
 241     long value;
 242 
 243     while (isspace ((unsigned char) *where))
 244     {                           /* Skip spaces */
 245         where++;
 246         if (--digs <= 0)
 247             return -1;          /* All blank field */
 248     }
 249     value = 0;
 250     while (digs > 0 && isodigit (*where))
 251     {                           /* Scan till nonoctal */
 252         value = (value << 3) | (*where++ - '0');
 253         --digs;
 254     }
 255 
 256     if (digs > 0 && *where && !isspace ((unsigned char) *where))
 257         return -1;              /* Ended on non-space/nul */
 258 
 259     return value;
 260 }

 283 tar_open_archive_int (struct vfs_class *me, const vfs_path_t * vpath, struct vfs_s_super *archive)
     /*  [+283 src/vfs/tar/tar.c] */
 284 {
 285     int result, type;
 286     tar_super_data_t *arch;
 287     mode_t mode;
 288     struct vfs_s_inode *root;
 289 
 290     result = mc_open (vpath, O_RDONLY);
 291     if (result == -1)
 292     {
 293         message (D_ERROR, MSG_ERROR, _("Cannot open tar archive\n%s"), vfs_path_as_str (vpath));
 294         ERRNOR (ENOENT, -1);
 295     }
 296 
 297     archive->name = g_strdup (vfs_path_as_str (vpath));
 298     archive->data = g_new (tar_super_data_t, 1);
 299     arch = (tar_super_data_t *) archive->data;
 300     mc_stat (vpath, &arch->st);
 301     arch->fd = -1;
 302     arch->type = TAR_UNKNOWN;
 303 
 304     /* Find out the method to handle this tar file */
 305     type = get_compression_type (result, archive->name);
 306     if (type == COMPRESSION_NONE)
 307         mc_lseek (result, 0, SEEK_SET);
 308     else
 309     {
 310         char *s;
 311         vfs_path_t *tmp_vpath;
 312 
 313         mc_close (result);
 314         s = g_strconcat (archive->name, decompress_extension (type), (char *) NULL);
 315         tmp_vpath = vfs_path_from_str_flags (s, VPF_NO_CANON);
 316         result = mc_open (tmp_vpath, O_RDONLY);
 317         vfs_path_free (tmp_vpath);
 318         if (result == -1)
 319             message (D_ERROR, MSG_ERROR, _("Cannot open tar archive\n%s"), s);
 320         g_free (s);
 321         if (result == -1)
 322         {
 323             MC_PTR_FREE (archive->name);
 324             ERRNOR (ENOENT, -1);
 325         }
 326     }
 327 
 328     arch->fd = result;
 329     mode = arch->st.st_mode & 07777;
 330     if (mode & 0400)
 331         mode |= 0100;
 332     if (mode & 0040)
 333         mode |= 0010;
 334     if (mode & 0004)
 335         mode |= 0001;
 336     mode |= S_IFDIR;
 337 
 338     root = vfs_s_new_inode (me, archive, &arch->st);
 339     root->st.st_mode = mode;
 340     root->data_offset = -1;
 341     root->st.st_nlink++;
 342     root->st.st_dev = MEDATA->rdev++;
 343 
 344     archive->root = root;
 345 
 346     return result;
 347 }

 378 static void
 379 tar_fill_stat (struct vfs_s_super *archive, struct stat *st, union record *header, size_t h_size)
     /*  [+379 src/vfs/tar/tar.c] */
 380 {
 381     tar_super_data_t *arch = (tar_super_data_t *) archive->data;
 382 
 383     st->st_mode = tar_from_oct (8, header->header.mode);
 384 
 385     /* Adjust st->st_mode because there are tar-files with
 386      * linkflag==LF_SYMLINK and S_ISLNK(mod)==0. I don't 
 387      * know about the other modes but I think I cause no new
 388      * problem when I adjust them, too. -- Norbert.
 389      */
 390     if (header->header.linkflag == LF_DIR)
 391         st->st_mode |= S_IFDIR;
 392     else if (header->header.linkflag == LF_SYMLINK)
 393         st->st_mode |= S_IFLNK;
 394     else if (header->header.linkflag == LF_CHR)
 395         st->st_mode |= S_IFCHR;
 396     else if (header->header.linkflag == LF_BLK)
 397         st->st_mode |= S_IFBLK;
 398     else if (header->header.linkflag == LF_FIFO)
 399         st->st_mode |= S_IFIFO;
 400     else
 401         st->st_mode |= S_IFREG;
 402 
 403     st->st_dev = 0;
 404     st->st_rdev = 0;
 405     switch (arch->type)
 406     {
 407     case TAR_USTAR:
 408     case TAR_POSIX:
 409     case TAR_GNU:
 410         /* *INDENT-OFF* */
 411         st->st_uid = *header->header.uname
 412             ? vfs_finduid (header->header.uname)
 413             : tar_from_oct (8, header->header.uid);
 414         st->st_gid = *header->header.gname
 415             ? vfs_findgid (header->header.gname)
 416             : tar_from_oct (8,header->header.gid);
 417         /* *INDENT-ON* */
 418 
 419         switch (header->header.linkflag)
 420         {
 421         case LF_BLK:
 422         case LF_CHR:
 423             st->st_rdev =
 424                 (tar_from_oct (8, header->header.devmajor) << 8) |
 425                 tar_from_oct (8, header->header.devminor);
 426         default:
 427             break;
 428         }
 429     default:
 430         st->st_uid = tar_from_oct (8, header->header.uid);
 431         st->st_gid = tar_from_oct (8, header->header.gid);
 432     }
 433     st->st_size = h_size;
 434     st->st_mtime = tar_from_oct (1 + 12, header->header.mtime);
 435     st->st_atime = 0;
 436     st->st_ctime = 0;
 437     if (arch->type == TAR_GNU)
 438     {
 439         st->st_atime = tar_from_oct (1 + 12, header->header.unused.oldgnu.atime);
 440         st->st_ctime = tar_from_oct (1 + 12, header->header.unused.oldgnu.ctime);
 441     }
 442 }

 445 /**
 446  * Return 1 for success, 0 if the checksum is bad, EOF on eof,
 447  * 2 for a record full of zeros (EOF marker).
 448  *
 449  */
 450 static ReadStatus
 451 tar_read_header (struct vfs_class *me, struct vfs_s_super *archive, int tard, size_t * h_size)
     /*  [+451 src/vfs/tar/tar.c] */
 452 {
 453     tar_super_data_t *arch = (tar_super_data_t *) archive->data;
 454 
 455     int i;
 456     long sum, signed_sum, recsum;
 457     char *p;
 458     union record *header;
 459     static char *next_long_name = NULL, *next_long_link = NULL;
 460 
 461   recurse:
 462 
 463     header = tar_get_next_record (archive, tard);
 464     if (NULL == header)
 465         return STATUS_EOF;
 466 
 467     recsum = tar_from_oct (8, header->header.chksum);
 468 
 469     sum = 0;
 470     signed_sum = 0;
 471     p = header->charptr;
 472     for (i = sizeof (*header); --i >= 0;)
 473     {
 474         /*
 475          * We can't use unsigned char here because of old compilers,
 476          * e.g. V7.
 477          */
 478         signed_sum += *p;
 479         sum += 0xFF & *p++;
 480     }
 481 
 482     /* Adjust checksum to count the "chksum" field as blanks. */
 483     for (i = sizeof (header->header.chksum); --i >= 0;)
 484     {
 485         sum -= 0xFF & header->header.chksum[i];
 486         signed_sum -= (char) header->header.chksum[i];
 487     }
 488     sum += ' ' * sizeof header->header.chksum;
 489     signed_sum += ' ' * sizeof header->header.chksum;
 490 
 491     /*
 492      * This is a zeroed record...whole record is 0's except
 493      * for the 8 blanks we faked for the checksum field.
 494      */
 495     if (sum == 8 * ' ')
 496         return STATUS_EOFMARK;
 497 
 498     if (sum != recsum && signed_sum != recsum)
 499         return STATUS_BADCHECKSUM;
 500 
 501     /*
 502      * Try to determine the archive format.
 503      */
 504     if (arch->type == TAR_UNKNOWN)
 505     {
 506         if (strcmp (header->header.magic, TMAGIC) == 0)
 507         {
 508             if (header->header.linkflag == LF_GLOBAL_EXTHDR)
 509                 arch->type = TAR_POSIX;
 510             else
 511                 arch->type = TAR_USTAR;
 512         }
 513         else if (strcmp (header->header.magic, OLDGNU_MAGIC) == 0)
 514             arch->type = TAR_GNU;
 515     }
 516 
 517     /*
 518      * linkflag on BSDI tar (pax) always '\000'
 519      */
 520     if (header->header.linkflag == '\000')
 521     {
 522         size_t len;
 523 
 524         if (header->header.arch_name[NAMSIZ - 1] != '\0')
 525             len = NAMSIZ;
 526         else
 527             len = strlen (header->header.arch_name);
 528 
 529         if (len != 0 && IS_PATH_SEP (header->header.arch_name[len - 1]))
 530             header->header.linkflag = LF_DIR;
 531     }
 532 
 533     /*
 534      * Good record.  Decode file size and return.
 535      */
 536     if (header->header.linkflag == LF_LINK || header->header.linkflag == LF_DIR)
 537         *h_size = 0;            /* Links 0 size on tape */
 538     else
 539         *h_size = tar_from_oct (1 + 12, header->header.size);
 540 
 541     /*
 542      * Skip over directory snapshot info records that
 543      * are stored in incremental tar archives.
 544      */
 545     if (header->header.linkflag == LF_DUMPDIR)
 546     {
 547         if (arch->type == TAR_UNKNOWN)
 548             arch->type = TAR_GNU;
 549         return STATUS_SUCCESS;
 550     }
 551 
 552     /*
 553      * Skip over pax extended header and global extended
 554      * header records.
 555      */
 556     if (header->header.linkflag == LF_EXTHDR || header->header.linkflag == LF_GLOBAL_EXTHDR)
 557     {
 558         if (arch->type == TAR_UNKNOWN)
 559             arch->type = TAR_POSIX;
 560         return STATUS_SUCCESS;
 561     }
 562 
 563     if (header->header.linkflag == LF_LONGNAME || header->header.linkflag == LF_LONGLINK)
 564     {
 565         char **longp;
 566         char *bp, *data;
 567         off_t size;
 568         size_t written;
 569 
 570         if (arch->type == TAR_UNKNOWN)
 571             arch->type = TAR_GNU;
 572 
 573         if (*h_size > MC_MAXPATHLEN)
 574         {
 575             message (D_ERROR, MSG_ERROR, _("Inconsistent tar archive"));
 576             return STATUS_BADCHECKSUM;
 577         }
 578 
 579         longp = ((header->header.linkflag == LF_LONGNAME) ? &next_long_name : &next_long_link);
 580 
 581         g_free (*longp);
 582         bp = *longp = g_malloc (*h_size + 1);
 583 
 584         for (size = *h_size; size > 0; size -= written)
 585         {
 586             data = tar_get_next_record (archive, tard)->charptr;
 587             if (data == NULL)
 588             {
 589                 MC_PTR_FREE (*longp);
 590                 message (D_ERROR, MSG_ERROR, _("Unexpected EOF on archive file"));
 591                 return STATUS_BADCHECKSUM;
 592             }
 593             written = RECORDSIZE;
 594             if ((off_t) written > size)
 595                 written = (size_t) size;
 596 
 597             memcpy (bp, data, written);
 598             bp += written;
 599         }
 600 
 601         if (bp - *longp == MC_MAXPATHLEN && bp[-1] != '\0')
 602         {
 603             MC_PTR_FREE (*longp);
 604             message (D_ERROR, MSG_ERROR, _("Inconsistent tar archive"));
 605             return STATUS_BADCHECKSUM;
 606         }
 607         *bp = 0;
 608         goto recurse;
 609     }
 610     else
 611     {
 612         struct stat st;
 613         struct vfs_s_entry *entry;
 614         struct vfs_s_inode *inode = NULL, *parent;
 615         off_t data_position;
 616         char *q;
 617         size_t len;
 618         char *current_file_name, *current_link_name;
 619 
 620         current_link_name =
 621             (next_long_link ? next_long_link : g_strndup (header->header.arch_linkname, NAMSIZ));
 622         len = strlen (current_link_name);
 623         if (len > 1 && IS_PATH_SEP (current_link_name[len - 1]))
 624             current_link_name[len - 1] = '\0';
 625 
 626         current_file_name = NULL;
 627         switch (arch->type)
 628         {
 629         case TAR_USTAR:
 630         case TAR_POSIX:
 631             /* The ustar archive format supports pathnames of upto 256
 632              * characters in length. This is achieved by concatenating
 633              * the contents of the 'prefix' and 'arch_name' fields like
 634              * this:
 635              *
 636              *   prefix + path_separator + arch_name
 637              *
 638              * If the 'prefix' field contains an empty string i.e. its
 639              * first characters is '\0' the prefix field is ignored.
 640              */
 641             if (header->header.unused.prefix[0] != '\0')
 642             {
 643                 char *temp_name, *temp_prefix;
 644 
 645                 temp_name = g_strndup (header->header.arch_name, NAMSIZ);
 646                 temp_prefix = g_strndup (header->header.unused.prefix, PREFIX_SIZE);
 647                 current_file_name = g_strconcat (temp_prefix, PATH_SEP_STR,
 648                                                  temp_name, (char *) NULL);
 649                 g_free (temp_name);
 650                 g_free (temp_prefix);
 651             }
 652             break;
 653         case TAR_GNU:
 654             if (next_long_name != NULL)
 655                 current_file_name = next_long_name;
 656             break;
 657         default:
 658             break;
 659         }
 660 
 661         if (current_file_name == NULL)
 662         {
 663             if (next_long_name != NULL)
 664                 current_file_name = g_strdup (next_long_name);
 665             else
 666                 current_file_name = g_strndup (header->header.arch_name, NAMSIZ);
 667         }
 668 
 669         canonicalize_pathname (current_file_name);

 743         return STATUS_SUCCESS;

 748 /**
 749  * Main loop for reading an archive.
 750  * Returns 0 on success, -1 on error.
 751  */
 752 static int
 753 tar_open_archive (struct vfs_s_super *archive, const vfs_path_t * vpath,
     /*  [+753 src/vfs/tar/tar.c] */
 754                   const vfs_path_element_t * vpath_element)
 755 {
 756     /* Initial status at start of archive */
 757     ReadStatus status = STATUS_EOFMARK;
 758     int tard;
 759 
 760     current_tar_position = 0;
 761     /* Open for reading */
 762     tard = tar_open_archive_int (vpath_element->class, vpath, archive);
 763     if (tard == -1)
 764         return -1;
 765 
 766     while (TRUE)
 767     {
 768         size_t h_size;
 769         ReadStatus prev_status = status;
 770 
 771         status = tar_read_header (vpath_element->class, archive, tard, &h_size);
 772 
 773         switch (status)
 774         {
 775         case STATUS_SUCCESS:
 776             tar_skip_n_records (archive, tard, (h_size + RECORDSIZE - 1) / RECORDSIZE);
 777             continue;
 778 
 779             /*
 780              * Invalid header:
 781              *
 782              * If the previous header was good, tell them
 783              * that we are skipping bad ones.
 784              */
 785         case STATUS_BADCHECKSUM:
 786             switch (prev_status)
 787             {
 788                 /* Error on first record */
 789             case STATUS_EOFMARK:
 790                 {
 791                     message (D_ERROR, MSG_ERROR, _("%s\ndoesn't look like a tar archive."),
 792                              vfs_path_as_str (vpath));
 793                     /* FALL THRU */
 794 
 795                     /* Error after header rec */
 796                 }
 797             case STATUS_SUCCESS:
 798                 /* Error after error */
 799 
 800             case STATUS_BADCHECKSUM:
 801                 return -1;
 802 
 803             case STATUS_EOF:
 804                 return 0;
 805 
 806             default:
 807                 break;
 808             }
 809 
 810             /* Record of zeroes */
 811         case STATUS_EOFMARK:   /* If error after 0's */
 812             /* FALL THRU */
 813             /* exit from loop */
 814         case STATUS_EOF:       /* End of archive */
 815             break;
 816         default:
 817             break;
 818         }
 819         break;
 820     }
 821     return 0;
 822 }

     /*  [+866 src/vfs/tar/tar.c] */

 874         ERRNOR (EIO, -1);

 878     res = mc_read (fd, buffer, count);
 879     if (res == -1)
 880         ERRNOR (errno, -1);

 882     FH->pos += res;
 883     return res;
