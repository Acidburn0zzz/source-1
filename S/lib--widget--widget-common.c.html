  62 struct hotkey_t
  63 parse_hotkey (const char *text)
     /*  [+63 lib/widget/widget-common.c] */
  64 {
  65     hotkey_t result;
  66     const char *cp, *p;
  67 
  68     if (text == NULL)
  69         text = "";
  70 
  71     /* search for '&', that is not on the of text */
  72     cp = strchr (text, '&');
  73     if (cp != NULL && cp[1] != '\0')
  74     {
  75         result.start = g_strndup (text, cp - text);
  76 
  77         /* skip '&' */
  78         cp++;
  79         p = str_cget_next_char (cp);
  80         result.hotkey = g_strndup (cp, p - cp);
  81 
  82         cp = p;
  83         result.end = g_strdup (cp);
  84     }
  85     else
  86     {
  87         result.start = g_strdup (text);
  88         result.hotkey = NULL;
  89         result.end = NULL;
  90     }
  91 
  92     return result;
  93 }

 209 /**
 210  * Modify state of widget.
 211  *
 212  * @param w      widget
 213  * @param state  widget state flag to modify
 214  * @param enable specifies whether to turn the flag on (TRUE) or off (FALSE).
 215  *               Only one flag per call can be modified.
 216  * @return       TRUE if set was handled successfully, FALSE otherwise
 217  */
 218 gboolean
 219 widget_set_state (Widget * w, widget_state_t state, gboolean enable)
     /*  [+219 lib/widget/widget-common.c] */
 220 {
 221     gboolean ret = TRUE;
 222 
 223     if (enable)
 224         w->state |= state;
 225     else
 226         w->state &= ~state;
 227 
 228     if (enable)
 229     {
 230         /* exclusive bits */
 231         if ((state & WST_CONSTRUCT) != 0)
 232             w->state &= ~(WST_ACTIVE | WST_SUSPENDED | WST_CLOSED);
 233         else if ((state & WST_ACTIVE) != 0)
 234             w->state &= ~(WST_CONSTRUCT | WST_SUSPENDED | WST_CLOSED);
 235         else if ((state & WST_SUSPENDED) != 0)
 236             w->state &= ~(WST_CONSTRUCT | WST_ACTIVE | WST_CLOSED);
 237         else if ((state & WST_CLOSED) != 0)
 238             w->state &= ~(WST_CONSTRUCT | WST_ACTIVE | WST_SUSPENDED);
 239     }
 240 
 241     if (w->owner == NULL)
 242         return FALSE;
 243 
 244     switch (state)
 245     {
 246     case WST_DISABLED:
 247         if (send_message (w, NULL, enable ? MSG_DISABLE : MSG_ENABLE, 0, NULL) != MSG_HANDLED)
 248             ret = FALSE;
 249         if (ret)
 250             send_message (w, NULL, MSG_DRAW, 0, NULL);
 251         break;
 252 
 253     default:
 254         break;
 255     }
 256 
 257     return ret;
 258 }

 292             color = h->color[DLG_COLOR_FOCUS];
 293         else
 294             color = h->color[DLG_COLOR_NORMAL];

 345 void
 346 widget_replace (Widget * old_w, Widget * new_w)
     /*  [+346 lib/widget/widget-common.c] */
 347 {
 348     WDialog *h = old_w->owner;
 349     gboolean should_focus = FALSE;
 350 
 351     if (h->widgets == NULL)
 352         return;
 353 
 354     if (h->current == NULL)
 355         h->current = h->widgets;
 356 
 357     if (old_w == h->current->data)
 358         should_focus = TRUE;
 359 
 360     new_w->owner = h;
 361     new_w->id = old_w->id;
 362 
 363     if (should_focus)
 364         h->current->data = new_w;
 365     else
 366         g_list_find (h->widgets, old_w)->data = new_w;
 367 
 368     send_message (old_w, NULL, MSG_DESTROY, 0, NULL);
 369     send_message (new_w, NULL, MSG_INIT, 0, NULL);
 370 
 371     if (should_focus)
 372         dlg_select_widget (new_w);
 373 
 374     widget_redraw (new_w);
 375 }

     /*  [+387 lib/widget/widget-common.c] */
